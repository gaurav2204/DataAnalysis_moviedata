# make a function name max_min in which can help out to find the max and min value of column enter value 
def max_min(column):
    #for maximum value
    max_value = df_movie[column].idxmax()
    maximum_value = pd.DataFrame(df_movie.loc[max_value])
    
    #for minimum value
    min_value = df_movie[column].idxmin()
    minimum_value = pd.DataFrame(df_movie.loc[min_value])
    
    #Both max and min value in one place 
    info = pd.concat([maximum_value , minimum_value ], axis=1)
    
    return info
